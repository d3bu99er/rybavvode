=== ПРОВЕРКА НАЛИЧИЯ ВАЛИДНОЙ КУКИ:
docker compose exec -T app python - <<'PY'
import asyncio
from app.config import get_settings
from app.services.forum_auth import ForumAuthService

async def main():
    s = get_settings()
    a = ForumAuthService(
        forum_root_url=s.forum_root_url,
        forum_login_url=s.forum_login_url,
        username=s.forum_username,
        password=s.forum_password,
        timeout_seconds=s.http_timeout_seconds,
        user_agent="diag/1.0",
        fallback_cookie=s.forum_session_cookie,
        preferred_cookie_name=s.forum_session_cookie_name,
    )
    c = await a.ensure_cookie(force_refresh=True)
    print("cookie_len:", len(c))
    print("has_xf_user:", "xf_user=" in c)

asyncio.run(main())
PY

=== ПРОВЕРКА КОЛИЧЕСТВА ПРОПУЩЕННЫХ НЕ СКАЧАННЫХ ВЛОЖЕНИЙ:
docker compose exec -T app python - <<'PY'
from sqlalchemy import select, func
from app.database import SessionLocal
from app.models import PostAttachment

db = SessionLocal()
total = db.scalar(select(func.count()).select_from(PostAttachment))
missing = db.scalar(select(func.count()).select_from(PostAttachment).where(PostAttachment.local_rel_path.is_(None)))
print("total:", total, "missing:", missing)
db.close()
PY

=== ЧТО ЗАНИМАЕТ МЕСТО?

#1) Общая картина по диску (топ папок на корне)
sudo du -xhd1 / | sort -h

#2) Что занимает место в проекте
cd /path/to/rybavvode
du -h --max-depth=2 . | sort -h | tail -n 50

#3) Что занимает Docker (образы/контейнеры/volumes/cache)
docker system df -v

#4) Размер volume с Postgres (обычно ..._pg_data)
PG_VOL=$(docker volume ls --format '{{.Name}}' | grep 'pg_data' | head -n1)
echo "PG_VOL=$PG_VOL"
sudo du -sh "$(docker volume inspect "$PG_VOL" -f '{{.Mountpoint}}')"

#5) Размер самой БД и крупнейших таблиц
docker compose exec db psql -U postgres -d fishingmap -c "SELECT pg_size_pretty(pg_database_size(current_database())) AS db_size;"
docker compose exec db psql -U postgres -d fishingmap -c "SELECT relname AS table_name, pg_size_pretty(pg_total_relation_size(relid)) AS total_size FROM pg_catalog.pg_statio_user_tables ORDER BY pg_total_relation_size(relid) DESC LIMIT 20;"
docker compose exec db psql -U postgres -d fishingmap -c "SELECT pg_size_pretty(SUM(size)) AS wal_size FROM pg_ls_waldir();"

#6) Самые большие docker-логи контейнеров
sudo find /var/lib/docker/containers -name '*-json.log' -printf '%10s %p\n' | sort -nr | head -n 20

#7) Системные логи
sudo du -sh /var/log/* 2>/dev/null | sort -h

==== ПОЧИСТИТЬ НЕНУЖНЫЕ ДОКЕР ОБРАЗЫ И ПРОЧЕЕ ГОВНО
# 1) Очистить build cache (самое важное в вашем случае)
docker builder prune -a -f

# 2) Удалить неиспользуемые образы
docker image prune -a -f

# 3) Дополнительно: остановленные контейнеры и лишние сети
docker container prune -f
docker network prune -f

# 4) Проверить итог
docker system df
